```typescript
import { z } from "zod";
import { and, eq } from "drizzle-orm";
import { mcpServer } from "../mcp.js";
import { db } from "../db/index.js";
import { words } from "../db/schema.js";

mcpServer.registerTool(
  "get_words_by_category",
  {
    description: "Get all active words for a given category, including register, frequency weight, and usage sentence.",
    inputSchema: z.object({
      category: z.string().describe("The category name (e.g. 'emphasis', 'summary')"),
    }),
    // No outputSchema — per research Pitfall 1, omitting avoids isError validation conflict
  },
  async ({ category }) => {
    try {
      const rows = await db
        .select({
          word: words.word,
          register: words.register,
          frequency: words.frequency,
          usageSentence: words.usageSentence,
        })
        .from(words)
        .where(and(eq(words.active, true), eq(words.category, category)));

      // Per D-03: human-readable text summary, NOT JSON serialization
      // Per Pitfall 5: handle null usageSentence with ?? "" to avoid "null" strings
      const summary = rows
        .map((r) => `${r.word} (${r.register}, freq:${r.frequency})`)
        .join(", ");

      const text = rows.length > 0
        ? `Words in '${category}': ${summary}`
        : `No active words found in category '${category}'`;

      return {
        content: [{ type: "text" as const, text }],
        structuredContent: {
          category,
          words: rows,
          count: rows.length,
        },
      };
    } catch (error) {
      const message = error instanceof Error ? error.message : "Unknown database error";
      return {
        isError: true,
        content: [{ type: "text" as const, text: `Failed to get words for category '${category}': ${message}` }],
      };
    }
  }
);
```

Key implementation details:
- Per D-03: text is `"Words in 'emphasis': therefore (formal, freq:80), hence (formal, freq:60)"` — human-readable.
- Per D-04: `structuredContent` is `{ category, words: [...], count }` — machine-parseable.
- Per MCP-02: returns `word`, `register`, `frequency`, and `usageSentence` fields.
- Per MCP-03: `and(eq(words.active, true), eq(words.category, category))` — only active words.
- Per MCP-04 (D-09): Both `content` and `structuredContent` in every success response.
- Per MCP-05 (D-10): try/catch with `isError: true` error path.

- Per Pitfall 5: `usageSentence` is nullable — it is passed through to `structuredContent` as-is (null is valid JSON), but NOT interpolated into the text summary string.
- No `outputSchema` defined — per research Pitfall 1.
- Uses `import { z } from "zod"` — Zod v3 from existing dependency.
